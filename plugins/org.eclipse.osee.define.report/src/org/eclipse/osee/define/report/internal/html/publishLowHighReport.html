Marc Potter Publish High/Low Requirement Report Publishes the High/Low Requirement Traceability Report
  • Selected Branch - provide the related Branch for the report
  • Select Types for report - Select any types listed in the drop down. Selected items will show in the list below
  •     Select one or more items in the selected list (using CTRL or SHIFT) and right click to remove

function oseeAppStart(oseeAppParams) { var branch = oseeAppParams["selected_branch"]; var csvTypes = oseeAppParams["selected_types"]; var uriTemplate = '##publishRequirementTemplate##'; var branchParam = '##branchParam##'; var selectedTypesParam = '##selectedTypesParam##'; var publishUri = uriTemplate.replace(branchParam, encodeURI(branch)); publishUri = publishUri.replace(selectedTypesParam, encodeURI(csvTypes)); var absoluteUri = window.location.href + "/" + publishUri; window.location.assign(absoluteUri); }
